157 static ExecutionStatus selectApplication(T_ProxyReader *inProxyReader, uint8_t *inAid, uint16_t inAidLength, uint8_t *outFci, uint16_t *outFciLength)
158 {
159  uint8_t command[K_APDU_MAX_LENGTH];
160  uint16_t commandLength = 0;
161  uint8_t response[K_APDU_MAX_LENGTH];
162  uint16_t responseLength = K_APDU_MAX_LENGTH;
163  ExecutionStatus status;
164 
165  // Pack Command
166  command[commandLength++] = 0x00; //CLA
167  command[commandLength++] = 0xA4; //INS
168  command[commandLength++] = 0x04; //P1 - Select Application
169  command[commandLength++] = 0x04; //P2 - Select first occurrence and return the FCI
170  command[commandLength++] = (uint8_t) (inAidLength & 0xFF); //Lc
171  OsKpl_Memcpy(command+commandLength, inAid, inAidLength); //DataIn
172  commandLength+= inAidLength;
173 
174 
175  status = inProxyReader->fProxyReaderTable->transmit(
176  inProxyReader,
177  command,
178  commandLength,
179  response,
180  &responseLength);
181 
182  if(status != EXECUTION_OK) {
183  return status;
184  }
185 
186  status = checkResponse(inProxyReader, response, &responseLength);
187 
188  if(status != EXECUTION_OK) {
189  return status;
190  }
191 
192  OsKpl_Memcpy(outFci, response, responseLength);
193  *outFciLength = responseLength;
194 
195  return EXECUTION_OK;
196 }
